#version 330 core

layout(location = 0) in vec3 position;
layout(location = 1) in vec2 texCoords;

out vec2 TexCrd;

uniform mat4 unifProjection = mat4(1.0);
uniform mat4 unifView = mat4(1.0);
uniform mat4 unifModel = mat4(1.0);

void main()
{
	gl_Position = unifProjection * unifView * unifModel * vec4(position, 1.0);
	TexCrd = texCoords;
}
