#version 330 core

layout(location = 0) out vec4 color;

in vertData
{
	vec4 position;
	vec4 vertColour;
	vec2 texCoords;
	float texID;
} inVertexData;

uniform sampler2D textures[32];

void main()
{
	vec4 pixelColor = inVertexData.vertColour;
	if(inVertexData.texID > 0.0)
	{
		pixelColor = texture(textures[int(inVertexData.texID - 0.5f)], inVertexData.texCoords);
		if(pixelColor.a < 0.1)
        discard;
	}

	color = pixelColor;
}
